SELECT COUNT(*)
FROM fenae_Participant p
INNER JOIN fenae_Company c
ON p.company_id = c.company_Id
WHERE c.company_Id = p.company_Id

SELECT p.function, COUNT(*) as count
FROM fenae_Participant p
INNER JOIN fenae_Company c
ON p.company_id = c.company_Id
WHERE c.company_Id = p.company_Id
GROUP BY p.function
